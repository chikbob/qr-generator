6️⃣  ОСНОВНЫЕ КОМАНДЫ ДЛЯ РАБОТЫ
────────────────────────────────────────────────────────────────────

  ЗАПУСК И ОСТАНОВКА:

    # Запустить все контейнеры
    docker-compose up -d

    # Остановить все контейнеры
    docker-compose down

    # Пересборка (если нужны изменения в Dockerfile)
    docker-compose build --no-cache

  РАЗРАБОТКА:

    # Просмотреть логи приложения
    docker-compose logs -f app

    # Войти в контейнер приложения (bash)
    docker-compose exec app bash

    # Войти в PHP консоль (Tinker)
    docker-compose exec app php artisan tinker

  БД И МИГРАЦИИ:

    # Запустить миграции
    docker-compose exec app php artisan migrate

    # Откат + заполнение данными
    docker-compose exec app php artisan migrate:fresh --seed

    # Доступ к MySQL консоли
    docker-compose exec db mysql -u root -psecret qr-generator-app

  ФРОНТЕНД:

    # Сборка для production
    docker-compose exec app npm run build

    # Development с hot reload (не останавливать контейнер)
    docker-compose exec app npm run dev

  ТЕСТИРОВАНИЕ:

    # Запустить все тесты
    docker-compose exec app php artisan test

    # Запустить конкретный тест
    docker-compose exec app php artisan test tests/Feature/HealthCheckTest.php
